-- criando BANCO DE DADOS

create database loja2023;

use loja2023;

CREATE TABLE CLIENTE(
CD_CLIENT INT NOT NULL PRIMARY KEY
NM_CLIENTE VARCHAR(30),
DS_CIDADE VARCHAR(30),
DS_UF VARCHAR(2)
);

CREATE TABLE PEDIDO (
CD_PEDIDO INT NOT NULL PRIMARY,
DT_PEDIDO DATATIME,
CD_CLIENTE INT
);

CREATE TABLE PRODUTO (
CD_PRODUTO INT NOT NULL PRIMARY KEY ,
DS_PRODUTO VARCHAR(50)
);

CREATE ITEM_PEDIDO(
CD_PEDIDO INT PRIMARY KEY,
CD_PRODUTO INT,
QT_PRODUTO INT
);
